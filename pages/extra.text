    <script>
        document.getElementById('payWithRazorpayBtn').addEventListener('click', function () {
            let amount = this.getAttribute("data-price");
            let user_id = this.getAttribute("data-user");
            let combo_id = this.getAttribute("data-combo");

            // Create order in Razorpay via PHP
            fetch('checkout.php', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
                body: new URLSearchParams({ amount, user_id, combo_id })
            })
                .then(response => response.json())
                .then(data => {
                    let options = {
                        "key": data.key,
                        "amount": amount * 100,
                        "currency": "INR",
                        "name": "Campus Corder",
                        "description": "Payment for Combo Offer",
                        "order_id": data.order_id,
                        "handler": function (response) {
                            savePayment(response, amount, user_id, combo_id);
                        }
                    };

                    let rzp = new Razorpay(options);
                    rzp.open();
                });
        });

        // Function to store payment details
        function savePayment(response, amount, user_id, combo_id) {
            fetch('save_payment.php', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
                body: new URLSearchParams({
                    payment_id: response.razorpay_payment_id,
                    order_id: response.razorpay_order_id,
                    amount: amount,
                    user_id: user_id,
                    combo_id: combo_id
                })
            })
                .then(res => res.json())
                .then(data => {
                    if (data.success) {
                        window.location.href = "success.php";
                    } else {
                        window.location.href = "cancel.php";
                        
                    }
                });
        }
    </script>
